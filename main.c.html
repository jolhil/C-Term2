#include <stdio.h>

#define SIZE 5

int calculateAverage(int grades[], int size[]) {
    int average;
    int total = 0;
    for (int i = 0; i < size[0]; i++){
        total = total + grades[i];
    }
    average = total / size[0];
    return average;
}

void printAverage(int grades[]) {
    int sizes[] = {SIZE};
    int average = calculateAverage(grades, sizes);
    int total = 0;
    for (int i = 0; i < SIZE; i++) {
        printf("Score %d is: %d\n", i + 1, grades[i]);
        total += grades[i];
    }
    printf("Total score is: %d\n", total);
    printf("Average is: %d\n", average);
}

int main() {
    int grades[SIZE] = {77, 83, 90, 100, 61};
    printAverage(grades);
    return 0;
}
